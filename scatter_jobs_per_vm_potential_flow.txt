workflow DistributeInputsToVMs {
    input {
        Array[String] inputs = ["input1", "input2", ..., "input100"]
    }
    
    Int inputsPerVM = 4
    Int totalVMs = length(inputs) / inputsPerVM
    
    scatter (vmIndex in range(totalVMs)) {
        Array[String] inputsForVM = slice(inputs, vmIndex * inputsPerVM, (vmIndex + 1) * inputsPerVM - 1)
        
        call RunTasksOnVM {
            input:
                inputsForVM = inputsForVM
        }
    }
}

task RunTasksOnVM {
    input {
        Array[String] inputsForVM
    }
    command {
        # Execute tasks for inputsForVM on the current VM
        for input in inputsForVM {
            # Run tasks on input
        }
    }
}
